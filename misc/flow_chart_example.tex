% Use the standalone class for displaying the tikz image on a small PDF.
\documentclass[crop, tikz]{standalone}

% Import the tikz package to use for the drawing.
\usepackage{tikz}

% The arrows package is used for the LaTeX arrow.
\usetikzlibrary{arrows.meta}

% Begin the document.
\begin{document}

    % Draw the figure.
    \begin{tikzpicture}[%
        > = Latex,
        line width = 0.2mm,
        line cap = round
    ]
        % Coordinates for the labels and nodes.
        \coordinate (A) at (+0.0, +4.0);
        \coordinate (B) at (+0.0, +0.0);
        \coordinate (C) at (-2.0, -4.0);
        \coordinate (D) at (+2.0, -5.0);
        \coordinate (E) at (+0.0, -5.0);

        % Arrows connecting nodes on the flow chart.
        \draw[->, shorten > = 0.25in] (A) to (B);
        \draw[->, shorten > = 0.40in] (B) to (E);

        % The three nodes in the flow chart.
        \draw[fill = white] (A) circle (1.2) node {Starting Thing};
        \draw[fill = white] (B) ellipse (1.2 and 0.6) node {Other Thing};
        \draw[fill = white] (C) rectangle node {The End Thing} (D);

        % Text that is not part of any of the nodes.
        \node (Text) at (4.0, 4.0) [below = 2in] {Super cool text};
    \end{tikzpicture}
\end{document}
