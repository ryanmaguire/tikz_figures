% Use the standalone class for displaying the tikz image on a small PDF.
\documentclass[crop, tikz]{standalone}

% Import the tikz and pgfplots packages to use for the drawing.
\usepackage{tikz, pgfplots}

% Begin the document.
\begin{document}

    % Draw the figure.
    \begin{tikzpicture}
        \begin{axis}[
            axis y line = center,
            axis x line = middle, 
            axis on top = true,
            xmin = -pi,
            xmax = pi,
            ymin = -3,
            ymax = 3,
            height = 12.0cm,
            width = 12.0cm,
            grid,
            xtick = {-5, ..., 5},
            ytick = {-2, -1, ..., 2},
            restrict y to domain = -3:3
        ]

            % pgfplots provides a plotting routines.
            \addplot [%
                domain = -pi:pi,
                samples = 1000,
                mark = none,
                ultra thick,
                blue
            ] {sin(deg(e^(x)))};

            % Label the plot.
            \node [above, blue] at (axis cs: 1.15,1.3) {$y=\sin(e^{x})$};
        \end{axis}
    \end{tikzpicture}
\end{document}
