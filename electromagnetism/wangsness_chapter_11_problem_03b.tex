% Use the standalone class for displaying the tikz image on a small PDF.
\documentclass[crop, tikz]{standalone}

% Import the tikz package to use for the drawing.
\usepackage{tikz}

% Needed for the minus symbol.
\usepackage{amssymb}

% Tikz packages used.
\usetikzlibrary{arrows.meta}

% Shorter minus symbol.
\DeclareMathSymbol{\minus}{\mathbin}{AMSa}{"39}

% Begin the document.
\begin{document}

    % Begin the drawing.
    \begin{tikzpicture}[> = Latex]

        % Draw the axes.
        \draw[<->, thick] (0.0, -4.3) to (0.0, 4.3) node[right] {$y$};
        \draw[<->, thick] (-4.3, 0.0) to (4.3, 0.0) node[above] {$x$};
        
        % Draw image charge in first quadrant.
        \draw[fill = black] (3.0, 2.0) circle (1.5pt);
        \draw[dashed] (0.0, 2.0) to node[above] {$a$} (3.0, 2.0);
        \draw[dashed] (3.0, 0.0) to node[right] {$b$} (3.0, 2.0);
        \node at (3.0, 2.0) [above right] {$q$};
        
        % Draw image charge in second quadrant.
        \draw[fill = black] (-3.0, 2.0) circle (1.5pt);
        \draw[dashed] (0.0, 2.0) to node[above] {$a$} (-3.0, 2.0);
        \draw[dashed] (-3.0, 0.0) to node[left] {$b$} (-3.0, 2.0);
        \node at (-3.0, 2.0) [above left] {$\minus{q}$};

        % Draw image charge in third quadrant.
        \draw[fill = black] (-3.0, -2.0) circle (1.5pt);
        \draw[dashed] (0.0, -2.0) to node[below] {$a$} (-3.0, -2.0);
        \draw[dashed] (-3.0, 0.0) to node[left] {$b$} (-3.0, -2.0);
        \node at (-3.0, -2.0) [below left] {$q$};

        % Draw image charge in first quadrant.
        \draw[fill = black] (3.0, -2.0) circle (1.5pt);
        \draw[dashed] (0.0, -2.0) to node[below] {$a$} (3.0, -2.0);
        \draw[dashed] (3.0, 0.0) to node[right] {$b$} (3.0, -2.0);
        \node at (3.0, -2.0) [below right] {$\minus{q}$};
    \end{tikzpicture}
\end{document}
