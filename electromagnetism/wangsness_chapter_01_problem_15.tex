% Use the standalone class for displaying the tikz image on a small PDF.
\documentclass[crop, tikz]{standalone}

% Import the tikz package to use for the drawing.
\usepackage{tikz}

% Packages used for decorating curves with arrows.
\usetikzlibrary{arrows.meta, decorations.markings}

% Begin the document.
\begin{document}

    % Begin the drawing.
    \begin{tikzpicture}[%
        > = Latex,
        line cap = round,
        ->-/.style = {% Marks a curve with an arrow in the middle.
            decoration = {%
                markings,
                mark = at position 0.55 with \arrow{Latex}
            },
            postaction = {decorate}
        },
        -<-/.style = {% Same idea, but in reverse direction.
            decoration={%
                markings,
                mark=at position .45 with \arrowreversed{Latex}
            },
            postaction = {decorate}
        }
    ]

        % Tick marks for the coordinate axes.
        \begin{scope}[line width = 0.2pt]
            \draw (1in, 0.03in) to (1in, -0.03in) node[below] {1};
            \draw (2in, 0.03in) to (2in, -0.03in) node[below] {2};
            \draw (0.03in, 2.0in) to (-0.03in, 2.0in) node[left] {$\sqrt{2k}$};
        \end{scope}

        % Draw the axes.
        \draw[->] (-0.1in, 0in) to (2.3in, 0in);
        \draw[->] (0in, -0.1in) to (0in, 2.5in);

        % Draw in the path for the line integral.
        \draw[->-, draw = blue] (0in, 0in) to (0in, 2in);
        \draw[->-, draw = blue] (0in, 2in) to (2in, 2in);
        \draw[rotate = 90, -<-, draw = blue] (0in, 0in) parabola (2in, -2in);

        % Label everything.
        \node at (1.0in, 1.0in) {$y^{2}=kx$};
        \node at (2.2in, 0.1in) {$x$};
        \node at (0.1in, 2.4in) {$y$};
    \end{tikzpicture}
\end{document}
