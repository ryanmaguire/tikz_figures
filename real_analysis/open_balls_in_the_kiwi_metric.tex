% Use the standalone class for displaying the tikz image on a small PDF.
\documentclass[crop, tikz]{standalone}

% Import the tikz package to use for the drawing.
\usepackage{tikz}

% Load the arrows.meta library.
\usetikzlibrary{arrows.meta}

% Begin the document.
\begin{document}

    % Draw the picture.
    \begin{tikzpicture}[%
        > = Latex,
        semithick,
        draw = blue,
        font = \scriptsize
    ]

        % Draw the first open ball.
        \begin{scope}[<->, draw = black]
            \draw (-1.8, 0.0) to (1.8, 0.0) node [above] {$x$};
            \draw (0.0, -1.5) to (0.0, 2.5) node [below right] {$y$};
        \end{scope}

        \draw[%
            fill = cyan,
            opacity = 0.8,
            densely dashed
        ] (-1.0, 0.0) to (0.0, -1.0) to (1.0, 0.0) to (0.0, 1.0) to cycle;

        % Draw the second open ball.
        \begin{scope}[<->, draw = black]
            \draw (2.2, 0.0) to (5.8, 0.0) node [above] {$x$};
            \draw (4.0, -1.5) to (4.0, 2.5) node [below right] {$y$};
        \end{scope}

        \draw (4.0, 0.0) to (4.0, 2.0);

        % Draw the third open ball.
        \begin{scope}[yshift = -4.5cm]
            \begin{scope}[<->, draw = black]
                \draw (-1.8, 0.0) to (1.8, 0.0) node [above] {$x$};
                \draw (0.0, -1.5) to (0.0, 2.5) node [below right] {$y$};
            \end{scope}

            \draw[%
                fill = cyan,
                opacity = 0.8,
                densely dashed
            ] (-0.5, 0.0) to (0.0, -0.5) to (0.5, 0.0) to (0.0, 0.5) to cycle;

            \draw (0.0, 0.5) to (0.0, 1.5);
        \end{scope}

        % Draw the fourth open ball.
        \begin{scope}[xshift = 4cm, yshift = -4.5cm]
            \begin{scope}[<->, draw = black]
                \draw (-1.8, 0.0) to (1.8, 0.0) node [above] {$x$};
                \draw (0.0, -1.5) to (0.0, 2.5) node [below right] {$y$};
            \end{scope}

            \draw[%
                fill = cyan,
                opacity = 0.8,
                densely dashed
            ] (0.0, 0.0) to (0.5, 0.5) to (1.0, 0.0) to (0.5, -0.5) to cycle;

            \draw (0.5, 0.5) to (0.5, 1.5);
        \end{scope}

        % Draw nodes indicating which points are not included.
        \begin{scope}[
            every node/.style = {
                circle,
                fill = white,
                draw = blue,
                inner sep = 0pt,
                minimum size = 2pt
            }
        ]
            \node at (4.0, 0.0) {};
            \node at (4.0, 2.0) {};
            \node at (0.0, -3.0) {};
            \node at (4.5, -3.0) {};
        \end{scope}
    \end{tikzpicture}
\end{document}
